
#include <ShiftregC1.h>
Shiftreg76 shiftRegister;
//LED PINS
#define SPI_LED 5
#define I2C_LED 6
#define PWM_LED 7

//I2C PINS
#define SDA 4
#define SCL 5

//CS PINS
#define P_CS1 2
#define P_CS2 1
#define P_CS3 0
#define C_CS 2

//SPI PINS
#define MISO_PIN 4







void setup(){
	
	//SETUP leds
	DDRD |= (1<<SPI_LED) | (1<<I2C_LED); //NO PWM LED ON S1

	//Initialize SPI
		S_SPI_INIT();
	//CLEAR SHIFT REGISTER
	shiftRegister.clearBoard();
}

void main(){
	//display recieved data in 
	shiftRegister.forwards(recieve());

}

byte recieve(){
	/* Wait for reception complete */
	while(!(SPSR & (1<<SPIF)))
	;
	/* Return Data Register */
	return SPDR;
}
void S_SPI_INIT(){
	//Set MISO output (MOSI input by default)
	PORTB |= (1<< MISO_PIN);
	//Enable SPI 
	SPCR = (1<<SPE);
}
